145 def aggiorna_blocchi_da_yaml(self, yaml_content):
146 """
147 @brief Parsea il file YAML e ricrea i blocchi dei sensori sul canvas.
148 """
149
150 # 1. Svuota tutti i blocchi esistenti
151 self.get_sensor_canvas().clear_blocks()
152
153 # 2. Parsea il contenuto YAML
154 yaml = YAML(typ="safe")
155 try:
156 data = yaml.load(yaml_content)
157 except Exception as e:
158 if hasattr(self, "logger"):
159 self.logger.log(f"{Translator.tr('yaml_parse_error')}: {e}", "error")
160 return
161
162 if not data or "sensor" not in data or not isinstance(data["sensor"], list):
163 if hasattr(self, "logger"):
164 self.logger.log(Translator.tr("no_sensor_section"), "warning")
165 return
166
167 # 3. Carica definizioni da sensors.json
168 base_dir = os.path.dirname(os.path.abspath(__file__))
169 json_path = os.path.join(base_dir, "..", "config", "sensors.json")
170 dialog = SensorSelectionDialog(sensors_json_path=json_path)
171 sensor_defs = dialog.sensors
172
173 # 4. Ricrea ogni blocco
174 for sensor in data["sensor"]:
175 platform = sensor.get("platform", "").lower()
176 name = sensor.get("name", "Nuovo Sensore")
177
178 # Trova definizione da JSON
179 sensor_def = next(
180 (s for s in sensor_defs if s["platform"] == platform or s["label"] == name),
181 None
182 )
183
184 if not sensor_def:
185 continue # salta sensori non definiti
186
187 # Crea blocco
188 blocco = SensorBlockItem(title=name)
189 blocco.name_edit.setText(name)
190
191 # Imposta tipo connessione
192 conn_type = self.tab_settings.detect_connection_type(sensor_def) if hasattr(self.tab_settings, "detect_connection_type") else "custom"
193 blocco.conn_type_display.setText(conn_type)
194
195 # Costruisci parametri e outputs
196 param_list = sensor_def.get("params", [])
197 blocco.build_from_params(param_list)
198
199 outputs = sensor_def.get("outputs", [])
200 blocco.build_from_returns(outputs)
201
202 # Popola i parametri dinamici dal YAML
203 for key, widget in blocco.param_widgets.items():
204 value = sensor.get(key)
205 if value is not None:
206 if isinstance(widget, QLineEdit):
207 widget.setText(str(value))
208 elif isinstance(widget, QSpinBox):
209 try:
210 widget.setValue(int(str(value).replace("s", "")))
211 except Exception:
212 pass
213 elif isinstance(widget, QComboBox):
214 idx = widget.findText(str(value))
215 if idx >= 0:
216 widget.setCurrentIndex(idx)
217
218 # Popola anche i campi output (es. temperature, humidity)
219 for output_key, output_widget in getattr(blocco, "output_links", {}).items():
220 if output_key in sensor and isinstance(sensor[output_key], dict):
221 name_val = sensor[output_key].get("name", "")
222 output_widget.setText(name_val)
223
224 self.get_sensor_canvas().add_sensor_block(blocco)
