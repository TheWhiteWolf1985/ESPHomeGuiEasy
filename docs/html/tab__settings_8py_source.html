183 def update_controller_image(self, board_value):
184 """
185 @brief Updates the displayed controller image based on the selected board.
186
187 Loads a PNG image from the assets/pinout folder matching the board's value,
188 and scales it proportionally to fit the display area.
189
190 If the image is not found, displays a fallback text.
191
192 @param board_value The internal value of the selected board (used as filename).
193 """
194 image_path = os.path.join("assets", "pinout", f"{board_value}.png")
195
196 label_width = self.controller_image.width()
197 if os.path.exists(image_path):
198 pixmap = QPixmap(image_path)
199 if label_width > 0:
200 scaled = pixmap.scaled(
201 label_width,
202 self.controller_image.height(),
203 Qt.AspectRatioMode.KeepAspectRatio,
204 Qt.TransformationMode.SmoothTransformation
205 )
206 self.controller_image.setPixmap(scaled)
207 self.controller_image.setText("")
208 else:
209 self.controller_image.setPixmap(pixmap)
210 self.controller_image.setText("")
211 else:
212 self.controller_image.setPixmap(QPixmap())
213 self.controller_image.setText(Translator.tr("image_not_available"))

244 def load_board_list(self):
245 """
246 @brief Loads the list of supported boards from a JSON file in the /config directory.
247
248 The expected JSON structure is:
249 {
250 "boards": [
251 {"label": "Commercial Name", "value": "internal_id"},
252 ...
253 ]
254 }
255
256 @return A list of dictionaries with 'label' and 'value' keys, sorted alphabetically.
257 """
258 try:
259 base_path = os.path.dirname(os.path.abspath(__file__))
260 config_path = os.path.join(base_path, "../config/boards.json")
261
262 with open(config_path, 'r', encoding='utf-8') as file:
263 data = json.load(file)
264 boards = data.get("boards", [])
265 # Ordina alfabeticamente per label (giusto in caso il file non lo sia già)
266 boards.sort(key=lambda x: x["label"].lower())
267 return boards
268 except (FileNotFoundError, json.JSONDecodeError) as e:
269 logger.error(f"[Errore] Caricamento boards.json fallito: {e}")
270 return []

272 def aggiorna_layout_da_dati(self):
273 """
274 @brief Updates the YAML content in the editor using only the general project fields.
275
276 Extracts device name, board, SSID and password from the tab inputs,
277 then regenerates the corresponding sections in the YAML editor.
278
279 @note This method does not modify the sensor section of the YAML.
280 """
281 main = self.window()
282 if not hasattr(main, "yaml_editor"):
283 if self.logger:
284 self.logger.log(Translator.tr("yaml_editor_not_found"), "error")
285 return
286
287 editor = main.yaml_editor
288 if editor is None:
289 if self.logger:
290 self.logger.log(Translator.tr("yaml_editor_is_none"), "error")
291 return
292
293 try:
294 current_yaml = editor.toPlainText()
295 device_name = self.get_device_name()
296 board_value = self.get_board()
297 ssid = self.get_ssid()
298 password = self.get_password()
299
300 new_yaml = YAMLHandler.generate_yaml_general_sections(
301 current_yaml=current_yaml,
302 device_name=device_name,
303 board=board_value,
304 ssid=ssid,
305 password=password
306 )
307 editor.setPlainText(new_yaml)
308 if self.logger:
309 self.logger.log(Translator.tr("yaml_updated_general"), "success")
310
311 except RuntimeError as e:
312 if self.logger:
313 self.logger.log(Translator.tr("yaml_editor_destroyed").format(error=str(e)), "error")

329 def carica_dati_da_yaml(self, yaml_content):
330 """
331 @brief Parses the YAML content and updates the Settings tab accordingly.
332
333 Extracts values from the YAML and populates:
334 - Device name
335 - Selected board
336 - Wi-Fi SSID and password
337
338 If the YAML is invalid or fields are missing, defaults are used.
339
340 @param yaml_content YAML content as a string.
341 """
342 yaml = YAML(typ="safe")
343 try:
344 data = yaml.load(yaml_content)
345 except Exception as e:
346 if hasattr(self, "logger"):
347 self.logger.log(Translator.tr("yaml_parse_error") + f": {e}", "error")
348 return
349
350 # Aggiorna i campi se presenti
351 if not data:
352 return
353 esphome = data.get("esphome", {})
354 self.device_name_edit.setText(esphome.get("name", ""))
355
356 esp32 = data.get("esp32", {})
357 board = esp32.get("board", "")
358 idx = self.board_combo.findData(board)
359 if idx >= 0:
360 self.board_combo.setCurrentIndex(idx)
361 else:
362 self.board_combo.setCurrentIndex(0)
363
364 wifi = data.get("wifi", {})
365 self.wifi_ssid_edit.setText(wifi.get("ssid", ""))
366 self.wifi_pass_edit.setText(wifi.get("password", ""))
