190 def load_board_list(self):
191 """
192 @brief Carica la lista delle board supportate da un file JSON in /config.
193
194 Il file deve avere una struttura:
195 {
196 "boards": [
197 {"label": "Nome commerciale", "value": "nome_tecnico"},
198 ...
199 ]
200 }
201
202 @return lista di dizionari con chiavi 'label' e 'value'
203 """
204 try:
205 base_path = os.path.dirname(os.path.abspath(__file__))
206 config_path = os.path.join(base_path, "../config/boards.json")
207
208 with open(config_path, 'r', encoding='utf-8') as file:
209 data = json.load(file)
210 boards = data.get("boards", [])
211 # Ordina alfabeticamente per label (giusto in caso il file non lo sia già)
212 boards.sort(key=lambda x: x["label"].lower())
213 return boards
214 except (FileNotFoundError, json.JSONDecodeError) as e:
215 logger.error(f"[Errore] Caricamento boards.json fallito: {e}")
216 return []

265 def carica_dati_da_yaml(self, yaml_content):
266 """
267 Legge lo YAML e aggiorna i campi del tab settings di conseguenza.
268 """
269 from ruamel.yaml import YAML
270 yaml = YAML(typ="safe")
271 try:
272 data = yaml.load(yaml_content)
273 except Exception as e:
274 if hasattr(self, "logger"):
275 self.logger.log(Translator.tr("yaml_parse_error") + f": {e}", "error")
276 return
277
278 # Aggiorna i campi se presenti
279 if not data:
280 return
281 esphome = data.get("esphome", {})
282 self.device_name_edit.setText(esphome.get("name", ""))
283
284 esp32 = data.get("esp32", {})
285 board = esp32.get("board", "")
286 idx = self.board_combo.findData(board)
287 if idx >= 0:
288 self.board_combo.setCurrentIndex(idx)
289 else:
290 self.board_combo.setCurrentIndex(0)
291
292 wifi = data.get("wifi", {})
293 self.wifi_ssid_edit.setText(wifi.get("ssid", ""))
