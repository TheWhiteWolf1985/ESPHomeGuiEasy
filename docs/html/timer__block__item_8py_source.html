152 def build_from_params(self, param_list):
153 """
154 @brief Builds the parameter input fields dynamically from a metadata list.
155
156 Accepts a list of dictionaries containing:
157 - key: unique parameter identifier
158 - type: one of ["text", "int", "combo"]
159 - label: field label
160 - default: default value
161 - options (if combo): available items
162
163 Supported widget types:
164 - QLineEdit for text
165 - QSpinBox for int
166 - QComboBox for combo
167
168 @param param_list List of parameter definitions in dictionary format.
169 """
170 layout = self.container.layout()
171 for param in param_list:
172 key = param.get("key")
173 tipo = param.get("type")
174 label_text = param.get("label", key)
175 default = param.get("default", "")
176
177 label = QLabel(label_text)
178 layout.addWidget(label)
179
180 if tipo == "text":
181 field = QLineEdit()
182 field.setPlaceholderText(param.get("placeholder", ""))
183 field.setText(str(default))
184 elif tipo == "int":
185 field = QSpinBox()
186 field.setMinimum(0)
187 field.setMaximum(100000)
188 if isinstance(default, int):
189 field.setValue(default)
190 elif tipo == "combo":
191 field = QComboBox()
192 options = param.get("options", [])
193 field.addItems(options)
194 if default in options:
195 field.setCurrentText(default)
196 else:
197 continue
198
199 layout.addWidget(field)
200 self.param_widgets[key] = field
