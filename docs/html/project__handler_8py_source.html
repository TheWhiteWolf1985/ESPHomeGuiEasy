27class ImportWorker(QObject):
28 """
29 @brief Qt worker class for importing ESPHome projects from a ZIP archive.
30
31 Extracts files to a destination folder and tries to locate a `.yaml` file to open automatically.
32
33 @signal progress(current: int, total: int): Emits progress percentage during extraction.
34 @signal finished(path_yaml: str, path_zip: str): Emits the YAML path once import is complete.
35 """
36 progress = pyqtSignal(int, int) # (current, total)
37 finished = pyqtSignal(str, str) # path_yaml, path_zip
38
39 def __init__(self, path_zip, path_dest):
40 super().__init__()
41 self.path_zip = path_zip
42 self.path_dest = path_dest
43
44 def run(self):
45 import os, zipfile
46 base_name = os.path.splitext(os.path.basename(self.path_zip))[0]
47 final_proj_dir = os.path.join(self.path_dest, base_name)
48 os.makedirs(final_proj_dir, exist_ok=True)
49
50 yaml_found = None
51 try:
52 with zipfile.ZipFile(self.path_zip, 'r') as zipf:
53 filelist = zipf.namelist()
54 total = len(filelist)
55 for i, member in enumerate(filelist, 1):
56 zipf.extract(member, final_proj_dir)
57 self.progress.emit(i, total)
58 for root, dirs, files in os.walk(final_proj_dir):
59 for file in files:
60 if file.endswith(".yaml") or file.endswith(".yml"):
61 yaml_found = os.path.join(root, file)
62 break
63 if yaml_found:
64 break
65 except Exception as e:
66 yaml_found = None
67 self.finished.emit(yaml_found or "", self.path_zip)

73class ExportWorker(QObject):
74 """
75 @brief Qt worker class for exporting a project folder to a compressed `.zip` archive.
76
77 Skips `.pioenvs` and emits signals for progress and completion.
78
79 @signal progress(current: int, total: int): Emits progress during compression.
80 @signal finished(path_zip: str): Emits the final ZIP file path.
81 """
82 progress = pyqtSignal(int, int) # (current, total)
83 finished = pyqtSignal(str) # path_zip
84
85 def __init__(self, project_dir, path_zip):
86 super().__init__()
87 self.project_dir = project_dir
88 self.path_zip = path_zip
89
90 def run(self):
91 filelist = []
92 for root, dirs, files in os.walk(self.project_dir):
93 for file in files:
94 abs_path = os.path.join(root, file)
95 rel_path = os.path.relpath(abs_path, self.project_dir)
96 if not rel_path.startswith('.pioenvs'):
97 filelist.append((abs_path, rel_path))
98 total = len(filelist)
99 try:
100 with zipfile.ZipFile(self.path_zip, 'w', zipfile.ZIP_DEFLATED) as zipf:
101 for i, (abs_path, rel_path) in enumerate(filelist, 1):
102 zipf.write(abs_path, rel_path)
103 self.progress.emit(i, total)
104 self.finished.emit(self.path_zip)
105 except Exception:
106 self.finished.emit("")
