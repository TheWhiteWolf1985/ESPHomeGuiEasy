152 def aggiorna_yaml_da_moduli(self):
153 """
154 @brief Updates the YAML content in the editor by reading the active module configurations.
155
156 Extracts module values from the UI widgets (accordion form),
157 generates a new YAML string using `YAMLHandler`, and replaces the current YAML in the editor.
158
159 If the operation is successful, a success message is logged.
160 On error, appropriate fallback logging is performed using the logger.
161
162 @note This method assumes that the editor is valid and the schema file is available.
163 """
164 try:
165 main = self.window()
166 if not hasattr(main, "yaml_editor") or main.yaml_editor is None:
167 return # editor non disponibile
168
169 editor = main.yaml_editor
170 from core.yaml_handler import YAMLHandler
171 modules_schema_path = "config/modules_schema.json"
172 current_yaml = editor.toPlainText()
173 modules_dict = YAMLHandler.extract_module_sections_from_widgets(
174 self.widget_map,
175 modules_schema_path
176 )
177 new_yaml = YAMLHandler.generate_yaml_with_modules(
178 current_yaml,
179 modules_dict,
180 modules_schema_path
181 )
182 editor.setPlainText(new_yaml)
183 if self.logger:
184 self.logger.log(Translator.tr("yaml_updated_from_modules"), "success")
185
186 except RuntimeError as e:
187 try:
188 if hasattr(self, "logger"):
189 self.logger.log(f"‚ùå YAML update crash: {e}", "error")
190 elif hasattr(main, "logger"):
191 main.logger.log(f"‚ùå YAML update crash: {e}", "error")
192 else:
193 logger.error(f"[TabModules] YAML update crash: {e}")
194 except Exception:
195 pass # Ultimo fallback

199 def carica_dati_da_yaml(self, yaml_string):
200 """
201 @brief Loads and parses YAML content, updating the UI module widgets accordingly.
202
203 For each module defined in the YAML, this method:
204 - Enables the module section (checkbox)
205 - Sets the correct values for fields like text, combo, checkbox, or spinbox
206
207 If a module is not found in the YAML, its section remains disabled.
208
209 @param yaml_string YAML configuration as a string to be parsed and reflected in the UI.
210 """
211 modules_schema_path = "config/modules_schema.json"
212 modules_data = YAMLHandler.extract_modules_from_yaml(
213 yaml_string,
214 modules_schema_path
215 )
216 # Estrai i moduli PRESENTI nel file yaml (chiavi minuscole, underscore)
217 import ruamel.yaml
218 yaml = ruamel.yaml.YAML()
219 data = yaml.load(yaml_string) or {}
220 yaml_moduli_presenti = set(data.keys())
221
222 for module_name, values in modules_data.items():
223 widget_dict = self.widget_map.get(module_name, {})
224 # Ricava la chiave yaml di questo modulo (debug, logger, ...)
225 yaml_key = module_name.lower().replace(" ", "_")
226 # --- GESTIONE CHECKBOX "ENABLED" ---
227 enabled_checkbox = widget_dict.get("enabled")
228 if isinstance(enabled_checkbox, QCheckBox):
229 enabled_checkbox.setChecked(yaml_key in yaml_moduli_presenti)
230 # --- Altri campi ---
231 for key, val in values.items():
232 widget = widget_dict.get(key)
233 if widget is None or key == "enabled":
234 continue
235 if isinstance(widget, QSpinBox):
236 if isinstance(val, str) and val.endswith("s"):
237 try:
238 val = int(val.rstrip("s"))
239 except Exception:
240 val = 1
241 try:
242 widget.setValue(int(val))
243 except Exception:
244 pass
245 elif isinstance(widget, QLineEdit):
246 widget.setText(str(val))
247 elif isinstance(widget, QComboBox):
248 idx = widget.findText(str(val))
249 if idx != -1:
250 widget.setCurrentIndex(idx)

252 def reset_fields(self):
253 """
254 @brief Resets all module widgets in the UI to their default states.
255
256 This includes:
257 - Clearing text fields
258 - Unchecking checkboxes
259 - Resetting combo boxes to the first item
260 - Setting spinboxes to their minimum value
261
262 It does not modify or clear the YAML editor.
263 """
264 for widget_dict in self.widget_map.values():
265 for key, widget in widget_dict.items():
266 if isinstance(widget, QLineEdit):
267 widget.clear()
268 elif isinstance(widget, QCheckBox):
269 # di default, puoi mettere checked/unchecked (scegli tu)
270 widget.setChecked(False)
271 elif isinstance(widget, QComboBox):
272 if widget.count() > 0:
273 widget.setCurrentIndex(0)
274 elif isinstance(widget, QSpinBox):
275 widget.setValue(widget.minimum())
