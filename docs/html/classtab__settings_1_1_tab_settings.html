190 def load_board_list(self):
191 """
192 @brief Carica la lista delle board supportate da un file JSON in /config.
193
194 Il file deve avere una struttura:
195 {
196 "boards": [
197 {"label": "Nome commerciale", "value": "nome_tecnico"},
198 ...
199 ]
200 }
201
202 @return lista di dizionari con chiavi 'label' e 'value'
203 """
204 try:
205 base_path = os.path.dirname(os.path.abspath(__file__))
206 config_path = os.path.join(base_path, "../config/boards.json")
207
208 with open(config_path, 'r', encoding='utf-8') as file:
209 data = json.load(file)
210 boards = data.get("boards", [])
211 # Ordina alfabeticamente per label (giusto in caso il file non lo sia già)
212 boards.sort(key=lambda x: x["label"].lower())
213 return boards
214 except (FileNotFoundError, json.JSONDecodeError) as e:
215 logger.error(f"[Errore] Caricamento boards.json fallito: {e}")
216 return []
