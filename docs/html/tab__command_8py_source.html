236 def scan_network_for_esp(self):
237 """Scansiona la rete locale per trovare ESPHome in ascolto sulla porta 3232."""
238 self.logger.log(Translator.tr("scan_in_progress"), "info")
239 self.ip_combo.clear()
240 found_ips = []
241
242 def scanner():
243 udp_port = 3232
244 message = b'ESPHomeDiscovery'
245 timeout = 2
246
247 # Invia un pacchetto broadcast UDP
248 s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM, socket.IPPROTO_UDP)
249 s.setsockopt(socket.SOL_SOCKET, socket.SO_BROADCAST, 1)
250 s.settimeout(timeout)
251 try:
252 s.sendto(message, ('<broadcast>', udp_port))
253 # Attendi le risposte
254 start = socket.getdefaulttimeout()
255 s.settimeout(1)
256 try:
257 while True:
258 data, addr = s.recvfrom(1024)
259 ip = addr[0]
260 if ip not in found_ips:
261 found_ips.append(ip)
262 self.logger.log(Translator.tr("device_found").format(ip=ip), "success")
263 # Aggiorna la combo (thread-safe)
264 self.ip_combo.addItem(ip)
265 except socket.timeout:
266 pass
267 except Exception as e:
268 self.logger.log(f"Errore scansione: {e}", "error")
269 finally:
270 s.close()
271 if not found_ips:
272 self.logger.log(Translator.tr("no_device_found"), "warning")
273 else:
274 self.logger.log(Translator.tr("devices_found").format(n=len(found_ips)), "success")
275
276 threading.Thread(target=scanner, daemon=True).start()
