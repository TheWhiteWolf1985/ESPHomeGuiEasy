279 def scan_network_for_esp(self):
280 """
281 @brief Scans the local network for ESPHome devices listening on UDP port 3232.
282
283 Sends a UDP broadcast and listens for replies within a timeout,
284 populates the IP combo box with discovered devices and logs status.
285 """
286 self.logger.log(Translator.tr("scan_in_progress"), "info")
287 self.ip_combo.clear()
288 found_ips = []
289
290 def scanner():
291 udp_port = 3232
292 message = b'ESPHomeDiscovery'
293 timeout = 2
294
295 # Invia un pacchetto broadcast UDP
296 s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM, socket.IPPROTO_UDP)
297 s.setsockopt(socket.SOL_SOCKET, socket.SO_BROADCAST, 1)
298 s.settimeout(timeout)
299 try:
300 s.sendto(message, ('<broadcast>', udp_port))
301 # Attendi le risposte
302 start = socket.getdefaulttimeout()
303 s.settimeout(1)
304 try:
305 while True:
306 data, addr = s.recvfrom(1024)
307 ip = addr[0]
308 if ip not in found_ips:
309 found_ips.append(ip)
310 self.logger.log(Translator.tr("device_found").format(ip=ip), "success")
311 # Aggiorna la combo (thread-safe)
312 self.ip_combo.addItem(ip)
313 except socket.timeout:
314 pass
315 except Exception as e:
316 self.logger.log(f"Errore scansione: {e}", "error")
317 finally:
318 s.close()
319 if not found_ips:
320 self.logger.log(Translator.tr("no_device_found"), "warning")
321 else:
322 self.logger.log(Translator.tr("devices_found").format(n=len(found_ips)), "success")
323
324 threading.Thread(target=scanner, daemon=True).start()
