28def create_new_project(data: dict, yaml_editor, logger, compiler, reset_tabs_callback, update_recent_callback):
29 """
30 @brief Creates a new project folder structure and updates the main GUI interface accordingly.
31
32 @param data Dictionary with input values from the project creation dialog.
33 @param yaml_editor Reference to the YAML editor widget to populate initial content.
34 @param logger Logging function or object with .log(msg, level) method.
35 @param compiler Object with set_project_dir(path) method.
36 @param reset_tabs_callback Function to reset UI tabs.
37 @param update_recent_callback Function to update the list of recent files.
38 @return Tuple (project_dir, yaml_path) if successful, or (None, None) on error.
39 """
40 nome_proj = data["name"].strip()
41 root_dir = Path(data["base_dir"]).expanduser()
42 categoria_raw = data.get("category", "Other / Misc").strip()
43 categoria = CATEGORY_TO_FOLDER.get(categoria_raw, "Other_Misc")
44 project_dir = root_dir / categoria / nome_proj
45
46
47 if project_dir.exists():
48 QMessageBox.warning(None, Translator.tr("warning"), Translator.tr("dir_exists"))
49 return None, None
50
51 try:
52 project_dir.mkdir(parents=True)
53 if data["create_subfolders"]:
54 (project_dir / "yaml").mkdir()
55 (project_dir / "logs").mkdir()
56 (project_dir / "assets").mkdir()
57 except Exception as e:
58 QMessageBox.critical(None, Translator.tr("error"), Translator.tr("create_dir_error").format(e=e))
59 return None, None
60
61 yaml_path = project_dir / f"{nome_proj}.yaml"
62
63 # === Generazione contenuto YAML compilato ===
64 try:
65 yaml_content = f"""esphome:
66 name: {nome_proj}
67 friendly_name: {nome_proj.replace("_", " ").title()}
68
69esp32:
70 board: esp32dev
71 framework:
72 type: arduino
73
74logger:
75
76api:
77
78ota:
79
80wifi:
81 ssid: "your_wifi"
82 password: "your_password"
83
84captive_portal:
85"""
86 with open(yaml_path, "w", encoding="utf-8") as f:
87 f.write(yaml_content)
88 except Exception as e:
89 QMessageBox.critical(None, Translator.tr("error"), f"Errore scrittura YAML: {e}")
90 return None, None
91
92 try:
93 with open(yaml_path, "r", encoding="utf-8") as f:
94 yaml_editor.setPlainText(f.read())
95 except Exception as e:
96 QMessageBox.critical(None, Translator.tr("error"), f"Errore apertura YAML: {e}")
97 return None, None
98
99 logger.log(Translator.tr("new_project_created").format(project_dir=project_dir), "success")
100
101 reset_tabs_callback()
102 compiler.set_project_dir(str(project_dir))
103 update_recent_callback()
104
105 info = {
106 "name": data["name"],
107 "author": data.get("author", "Unknown"),
108 "version": data.get("version", "1.0"),
109 "update": datetime.today().strftime("%Y-%m-%d"),
110 "category": data.get("category", "Other / Misc"),
111 "description": data.get("description", ""),
112 "changelog": [] # ✅ aggiunto per supporto incrementale
113 }
114
115 info_path = project_dir / "info.json"
116 try:
117 with open(info_path, "w", encoding="utf-8") as f:
118 json.dump(info, f, indent=2, ensure_ascii=False)
119 except Exception as e:
120 logger.log(f"❌ Errore nella creazione di info.json: {e}", "error")
121
122 return str(project_dir), str(yaml_path)
