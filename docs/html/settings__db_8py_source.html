35def init_db():
36 """
37 @brief Initializes the SQLite database if not already present.
38
39 Creates `settings` and `recent_files` tables if they do not exist.
40 Does not insert any default values.
41 """
42 conn = sqlite3.connect(get_user_db_path())
43 cursor = conn.cursor()
44 cursor.execute("""
45 CREATE TABLE IF NOT EXISTS settings (
46 key TEXT PRIMARY KEY,
47 value TEXT NOT NULL
48 )
49 """)
50 cursor.execute("""
51 CREATE TABLE IF NOT EXISTS recent_files (
52 path TEXT PRIMARY KEY,
53 filename TEXT NOT NULL,
54 last_opened TIMESTAMP DEFAULT CURRENT_TIMESTAMP
55 )
56 """)
57 conn.commit()
58 conn.close()

99def add_recent_file(path: str):
100 """
101 @brief Adds or updates a recently opened file in the `recent_files` table.
102
103 @param path Absolute path to the YAML file.
104 """
105 filename = os.path.basename(path)
106 conn = sqlite3.connect(get_user_db_path())
107 cursor = conn.cursor()
108 cursor.execute("""
109 INSERT INTO recent_files (path, filename, last_opened)
110 VALUES (?, ?, CURRENT_TIMESTAMP)
111 ON CONFLICT(path) DO UPDATE SET last_opened=CURRENT_TIMESTAMP, filename=excluded.filename
112 """, (path, filename))
113 conn.commit()
114 conn.close()
