34def format_changelog(changelog: list[dict]) -> str:
35 if not changelog:
36 return Translator.tr("no_changelog_available") # o ""
37
38 lines = []
39 for entry in reversed(changelog): # mostriamo prima i più recenti
40 version = entry.get("version", "?.?")
41 date = entry.get("date", "???")
42 text = entry.get("text", "").strip()
43
44 lines.append(f"🟢 Versione {version} ({date})")
45 if text:
46 lines.append(f"- {text}")
47 lines.append("") # riga vuota tra le voci

186 def load_project_metadata(self):
187 """
188 @brief Scans the user projects directory and reads metadata from each `info.json`.
189
190 For each subfolder under each category in `DEFAULT_PROJECT_DIR`, loads:
191 - name
192 - author
193 - version
194 - update
195 - description
196 - changelog (optional)
197
198 @return List of dictionaries, each containing one project's metadata.
199 """
200 projects = []
201 if not DEFAULT_PROJECT_DIR.exists():
202 return projects
203
204 for category in DEFAULT_PROJECT_DIR.iterdir():
205 if not category.is_dir():
206 continue
207 for project in category.iterdir():
208 info_path = project / "info.json"
209 if info_path.exists():
210 try:
211 with open(info_path, "r", encoding="utf-8") as f:
212 data = json.load(f)
213 data["__path"] = str(project)
214 data["category"] = category.name
215 projects.append(data)
216 except Exception as e:
217 print(f"Errore caricamento {info_path}: {e}")
218 return projects

368 def modifica_progetto(self, project_data):
369 """
370 @brief Opens the edit dialog to change project version and description.
371
372 After confirmation, updates:
373 - info.json
374 - update timestamp
375 - changelog array
376
377 Refreshes project list and card display.
378
379 @param project_data Dictionary with current project info.
380 """
381 info_path = Path(project_data.get("__path", "")) / "info.json"
382 if not info_path.exists():
383 self.show_message(Translator.tr("error"), "File info.json non trovato.")
384 return
385
386 try:
387 with open(info_path, "r", encoding="utf-8") as f:
388 data = json.load(f)
389 except Exception as e:
390 self.show_message(Translator.tr("error"), f"{Translator.tr('read_info_error')}\n{e}")
391 return
392
393 dialog = ProjectEditDialog(
394 version=data.get("version", ""),
395 description=data.get("description", ""),
396 parent=self
397 )
398
399 if dialog.exec() != QDialog.DialogCode.Accepted:
400 return
401
402 version, descrizione = dialog.get_data()
403
404 # Aggiorna dati principali
405 data["version"] = version
406 data["description"] = descrizione
407 data["update"] = datetime.today().strftime("%Y-%m-%d")
408
409 # Aggiorna changelog incrementale
410 changelog_entry = {
411 "date": data["update"],
412 "version": data["version"],
413 "text": data["description"]
414 }
415
416 if "changelog" not in data or not isinstance(data["changelog"], list):
417 data["changelog"] = []
418
419 data["changelog"].append(changelog_entry)
420
421 try:
422 with open(info_path, "w", encoding="utf-8") as f:
423 json.dump(data, f, indent=2, ensure_ascii=False)
424
425 self.show_message(Translator.tr("saved"), Translator.tr("save_success"))
426 self.project_data = self.load_project_metadata()
427 self.category_to_cards = self.build_category_index()
428 self.load_category_cards(project_data["category"])
429 except Exception as e:
430 self.show_message(Translator.tr("error"), Translator.tr("save_error").format(e=e))
