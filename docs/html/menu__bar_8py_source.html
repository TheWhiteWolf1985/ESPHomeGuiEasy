163 def _update_recent_files_menu(self):
164 for act in self.recent_file_actions:
165 self.file_menu.removeAction(act)
166 self.recent_file_actions.clear()
167
168 recent_files = get_recent_files(limit=4)
169 if recent_files:
170 self.file_menu.addSeparator()
171 for path, name in recent_files:
172 action = QAction(name, self)
173 action.setToolTip(path)
174 action.triggered.connect(self._make_open_file_handler(path))
175 self.file_menu.addAction(action)
176 self.recent_file_actions.append(action)
