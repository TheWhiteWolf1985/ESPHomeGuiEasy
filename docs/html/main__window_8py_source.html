272 def open_project(self, yaml_path):
273 """
274 @brief Loads an existing YAML project.
275
276 Receives the full path of a YAML file, reads its content, and inserts it into the editor.
277 Then synchronizes all configuration tabs (settings, modules, sensors) with the loaded data,
278 allowing visual editing of all sections.
279
280 Handles file read errors and logs them with error severity.
281 Updates current project path and saves the project opening history.
282 """
283 yaml_path = os.path.abspath(yaml_path)
284 self._reset_tabs()
285
286 try:
287 with open(yaml_path, "r", encoding="utf-8") as f:
288 content = f.read()
289 self.yaml_editor.setPlainText(content)
290 self.logger.log(f"Progetto caricato: {yaml_path}", "info")
291 except Exception as e:
292 self.logger.log(Translator.tr("error") + f": {e}", "error")
293 return
294
295 # 💡 PATCH CRUCIALE: forza aggiornamento come in importa_yaml()
296 print("[DEBUG] Avvio aggiornamento tab_settings")
297 self.tab_settings.carica_dati_da_yaml(content)
298
299 print("[DEBUG] Avvio aggiornamento tab_sensori")
300 self.tab_sensori.aggiorna_blocchi_da_yaml(content)
301
302 print("[DEBUG] Avvio aggiornamento tab_modules")
303 self.tab_modules.carica_dati_da_yaml(content)
304
305
306 self.logger.log(Translator.tr("project_opened").format(path=yaml_path), "success")
307 self.last_save_path = yaml_path
308 self.project_dir = os.path.dirname(yaml_path)
309 add_recent_file(yaml_path)
310 self.menu_bar._update_recent_files_menu()

469 def get_or_create_yaml_path(self) -> str:
470 """
471 @brief Determines the YAML file path to use for compilation or upload.
472
473 If the project has been saved, updates the existing file.
474 Otherwise, creates a temporary file in a dedicated build folder.
475
476 This manages both saved projects and unsaved changes without data loss.
477 """
478 yaml_text = self.yaml_editor.toPlainText()
479
480 if self.last_save_path:
481 with open(self.last_save_path, "w", encoding="utf-8") as f:
482 f.write(yaml_text)
483 self.logger.log(Translator.tr("file_saved_to").format(path=self.last_save_path), "success")
484 return self.last_save_path
485
486 # Se c'è un progetto, salva direttamente lì
487 if self.project_dir:
488 base_dir = self.project_dir
489 temp_dir = os.path.join(base_dir, ".temp")
490 os.makedirs(temp_dir, exist_ok=True)
491 temp_path = os.path.join(temp_dir, "__temp_upload.yaml")
492 else:
493 # Usa la cartella build centralizzata
494 temp_path = os.path.join(conf.DEFAULT_BUILD_DIR, "__temp_upload.yaml")
495
496 os.makedirs(temp_dir, exist_ok=True)
497 temp_path = os.path.join(temp_dir, "__temp_upload.yaml")
498
499 with open(temp_path, "w", encoding="utf-8") as f:
500 f.write(yaml_text)
501
502 self.logger.log(Translator.tr("unsaved_project_temp_warning"), "warning")
503 self.logger.log(Translator.tr("temp_file_generated").format(temp_path=temp_path), "info")
504 return temp_path
