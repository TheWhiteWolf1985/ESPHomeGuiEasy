28class SensorSelectionDialog(QDialog):
29 """
30 @brief Dialog window for selecting a sensor from the sensors JSON list.
31
32 Supports search filtering and double-click/OK selection.
33 """
34 def __init__(self, sensors_json_path: str, parent=None):
35 super().__init__(parent)
36 self.setWindowTitle("Seleziona Sensore")
37 self.setMinimumSize(400, 500)
38 self.selected_sensor = None
39
40 self.setStyleSheet(Pantone.DIALOG_STYLE)

42 self.sensors = self.load_sensors(sensors_json_path)
43
44 layout = QVBoxLayout(self)

46 # Filtro di ricerca
47 self.search_bar = QLineEdit()

49 self.search_bar.textChanged.connect(self.filter_list)
50 layout.addWidget(self.search_bar)
51
52 # Lista sensori
53 self.list_widget = QListWidget()
54 self.list_widget.setStyleSheet(Pantone.LISTWIDGET_STYLE)
55 self.list_widget.itemDoubleClicked.connect(self.accept_selection)
56 layout.addWidget(self.list_widget)

58 # Bottoni OK / Annulla

61 self.button_box.rejected.connect(self.reject)

66 def load_sensors(self, path):
67 try:
68 with open(path, "r", encoding="utf-8") as f:
69 return json.load(f).get("sensors", [])
70 except Exception as e:
71 logger.error(f"Errore nel caricamento dei sensori: {e}")
72 return []
73
74 def populate_list(self):
75 self.list_widget.clear()
76 for sensor in self.sensors:
77 label = sensor.get("label", "Sconosciuto")
78 platform = sensor.get("platform", "?")
79 conn_type = self.detect_connection_type(sensor)
80 item = QListWidgetItem(QIcon(self.get_icon_path(platform)), f"{label} ({conn_type})")
81 item.setData(Qt.ItemDataRole.UserRole, sensor)
82 self.list_widget.addItem(item)
83
84 def detect_connection_type(self, sensor):
85 # Determina la connessione in base ai parametri
86 keys = [p["key"] for p in sensor.get("params", [])]
87 if any("i2c" in k.lower() for k in keys):
88 return "I2C"
89 elif "analog" in sensor.get("label", "").lower():
90 return "Analogico"
91 elif "gpio" in sensor.get("platform", "").lower():
92 return "Digitale"
93 return "Sconosciuto"
94
95 def get_icon_path(self, platform):
96 # Supponiamo di avere le icone in assets/icons/{platform}.png
97 base_dir = os.path.dirname(os.path.abspath(__file__))
98 icon_path = os.path.join(base_dir, "..", "assets", "icons", f"{platform.lower()}.png")
99 if os.path.exists(icon_path):
100 return icon_path
101 return os.path.join(base_dir, "..", "assets", "icons", "default.png")
102
103 def filter_list(self, text):
104 for i in range(self.list_widget.count()):
105 item = self.list_widget.item(i)
106 item.setHidden(text.lower() not in item.text().lower())
107
108 def accept_selection(self):
109 item = self.list_widget.currentItem()
110 if item:
111 self.selected_sensor = item.data(Qt.ItemDataRole.UserRole)
112 self.accept()
113
114 def get_selected_sensor(self):
115 return self.selected_sensor
