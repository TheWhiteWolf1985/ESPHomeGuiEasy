59 item = QListWidgetItem(QIcon(self.get_icon_path(platform)), f"{label} ({conn_type})")

63 def detect_connection_type(self, sensor):
64 # Determina la connessione in base ai parametri
65 keys = [p["key"] for p in sensor.get("params", [])]
66 if any("i2c" in k.lower() for k in keys):
67 return "I2C"
68 elif "analog" in sensor.get("label", "").lower():
69 return "Analogico"
70 elif "gpio" in sensor.get("platform", "").lower():
71 return "Digitale"
72 return "Sconosciuto"

78 if os.path.exists(icon_path):
79 return icon_path
80 return os.path.join(base_dir, "..", "assets", "icons", "default.png")
