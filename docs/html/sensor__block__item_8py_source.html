178 def build_from_params(self, param_list):
179 """
180 @brief Aggiunge dinamicamente i campi dal JSON al layout del blocco.
181 @param param_list Lista di dizionari con i parametri (da sensors.json)
182 """
183 self.param_widgets = {}
184
185 layout = self.container.layout()
186 layout.addSpacing(10)
187
188 for param in param_list:
189 key = param.get("key")
190 tipo = param.get("type")
191 label_text = param.get("label", key)
192 default = param.get("default", "")
193 required = param.get("required", False)
194
195 label = QLabel(label_text)
196 layout.addWidget(label)
197
198 if tipo == "text":
199 field = QLineEdit()
200 field.setPlaceholderText(param.get("placeholder", ""))
201 field.setText(str(default))
202 layout.addWidget(field)
203
204 elif tipo == "int":
205 field = QSpinBox()
206 field.setMinimum(0)
207 field.setMaximum(100000)
208 if isinstance(default, int):
209 field.setValue(default)
210 layout.addWidget(field)
211
212 elif tipo == "combo":
213 field = QComboBox()
214 options = param.get("options", [])
215 field.addItems(options)
216 if default in options:
217 field.setCurrentText(default)
218 layout.addWidget(field)
219
220 else:
221 continue # Tipo non gestito (pu√≤ essere esteso)
222
223 self.param_widgets[key] = field # Salva il riferimento
224 self.required_fields[key] = required

255 def has_valid_data(self) -> bool:
256 """
257 @brief Verifica che tutti i campi richiesti siano stati compilati.
258 Evidenzia i campi mancanti in rosso.
259 @return True se validi, False altrimenti.
260 """
261 valid = True
262
263 for key, widget in self.param_widgets.items():
264 is_required = self.required_fields.get(key, False)
265
266 value = None
267 if isinstance(widget, QLineEdit):
268 value = widget.text().strip()
269 elif isinstance(widget, QComboBox):
270 value = widget.currentText().strip()
271 elif isinstance(widget, QSpinBox):
272 value = widget.value()
273
274 is_invalid = (
275 is_required and (value in ["", None] or (isinstance(value, int) and value == 0))
276 )
277
278 # Colora in rosso se mancante, altrimenti stile normale
279 if is_invalid:
280 widget.setStyleSheet("background-color: #ffcccc; color: #000; border: 1px solid #a00; border-radius: 4px;")
281 valid = False
282 else:
283 # Reset stile (stile scuro coerente)
284 widget.setStyleSheet("background-color: #2a2d2e; color: #d4d4d4; border: 1px solid #444; border-radius: 5px;")
285
286 return valid
