87 def line_number_area_size(self):
88 """
89 @brief Calculates the width needed to display all line numbers.
90
91 Depends on the number of digits in the total line count.
92
93 @return QSize with calculated width.

95 digits = len(str(max(1, self.blockCount())))
96 space = 3 + self.fontMetrics().horizontalAdvance('9') * digits

136 def line_number_area_paint(self, event):
137 """
138 @brief Custom paint method for rendering line numbers.
139
140 Iterates over visible text blocks and draws their line numbers
141 in the left margin.
142
143 @param event Paint event describing the updated region.
144 """
145 painter = QPainter(self.line_number_area)
146 painter.fillRect(event.rect(), Qt.GlobalColor.darkGray)
147
148 block = self.firstVisibleBlock()
149 block_number = block.blockNumber()
150 top = int(self.blockBoundingGeometry(block).translated(self.contentOffset()).top())
151 bottom = top + int(self.blockBoundingRect(block).height())
152
153 while block.isValid() and top <= event.rect().bottom():
154 if block.isVisible() and bottom >= event.rect().top():
155 number = str(block_number + 1)
156 painter.setPen(Qt.GlobalColor.white)
157 painter.drawText(0, top, self.line_number_area.width() - 4, self.fontMetrics().height(),
158 Qt.AlignmentFlag.AlignRight, number)
159 block = block.next()
160 top = bottom
161 bottom = top + int(self.blockBoundingRect(block).height())
162 block_number += 1
