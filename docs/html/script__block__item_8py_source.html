122 def build_from_params(self, param_list):
123 """
124 @brief Dynamically builds parameter input fields from a list of parameter definitions.
125
126 @param param_list List of dictionaries defining keys, types, labels, defaults, and options.
127 """
128 layout = self.container.layout()
129 for param in param_list:
130 key = param.get("key")
131 tipo = param.get("type")
132 label_text = param.get("label", key)
133 default = param.get("default", "")
134
135 label = QLabel(label_text)
136 layout.addWidget(label)
137
138 if tipo == "text":
139 field = QLineEdit()
140 field.setPlaceholderText(param.get("placeholder", ""))
141 field.setText(str(default))
142 elif tipo == "int":
143 field = QSpinBox()
144 field.setMinimum(0)
145 field.setMaximum(100000)
146 if isinstance(default, int):
147 field.setValue(default)
148 elif tipo == "combo":
149 field = QComboBox()
150 options = param.get("options", [])
151 field.addItems(options)
152 if default in options:
153 field.setCurrentText(default)
154 else:
155 continue
156
157 layout.addWidget(field)
158 self.param_widgets[key] = field
