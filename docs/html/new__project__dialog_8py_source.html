168 def validate_and_accept(self):
169 name = self.name_input.text().strip()
170 base_dir = self.base_dir_input.text().strip()
171
172 if not name:
173 QMessageBox.warning(self, Translator.tr("warning"), Translator.tr("error_project_name_required"))
174 return
175
176 # Controlla caratteri non validi per nomi di cartelle
177 if re.search(r'[<>:"/\\|?*]', name):
178 QMessageBox.warning(self, Translator.tr("warning"), Translator.tr("error_invalid_characters"))
179 return
180
181 # Fallback se base_dir è vuoto
182 if not base_dir:
183 self.base_dir_input.setText(str(DEFAULT_PROJECT_DIR))
184
185 self.accept()
