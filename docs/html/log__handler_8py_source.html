30class LOGHandler:
31 """
32@brief Outputs messages to a QTextEdit widget, with color formatting per log level.

40 @property
41 def console(self):
42 # Cerca un QTextEdit attivo con nome console_output
43 for w in QApplication.topLevelWidgets():
44 if hasattr(w, "console_output"):
45 return w.console_output
46 return self._console # fallback se non troviamo nulla

48 def log(self, message, level="info"):
49 console = self.console
50 if console is None:
51 print(f"[{level.upper()}] {message}")
52 return
53
54 try:
55 cursor = console.textCursor()
56 if level == "error":
57 color = "#ff5555"
58 elif level == "warning":
59 color = "#f1c40f"
60 elif level == "success":
61 color = "#33ff99"
62 else:
63 color = "#d4d4d4"
64
65 html = f'<span style="color:{color};">[{level.upper()}]</span> {message}<br>'
66 console.moveCursor(QTextCursor.MoveOperation.End)
67 console.insertHtml(html)
68 console.moveCursor(QTextCursor.MoveOperation.End)
69 except RuntimeError:
70 print(f"[LOGGER ERROR] QTextEdit distrutto. Messaggio:\n[{level.upper()}] {message}")
